$$
\begin{array}{l} \operatorname {C l} _ {2} (\mathrm {g}) = 2 \operatorname {C l} (\mathrm {g}) \quad K _ {p 1} = 0. 5 7 6 \\ \mathrm {H} _ {2} (\mathrm {g}) + \mathrm {C l} _ {2} (\mathrm {g}) = 2 \mathrm {H C l} (\mathrm {g}) K _ {p 2} \\ \mathrm {H C l} (\mathrm {g}) = \mathrm {H} (\mathrm {g}) + \mathrm {C l} (\mathrm {g}) \quad K _ {p 3} = 2. 2 3 \times 1 0 ^ {- 6} \\ \mathrm {H} _ {2} (\mathrm {g}) = 2 \mathrm {H} (\mathrm {g}) \quad K _ {p 4} \\ \end{array}
